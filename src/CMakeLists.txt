cmake_minimum_required(VERSION 3.10)

project(pinetime VERSION 1.13.0 LANGUAGES C CXX ASM)

set(CMAKE_C_STANDARD 99)
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_C_EXTENSIONS OFF)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

set(NRF_TARGET "nrf52")

# Toolchain and SDK paths validation
if (NOT ARM_NONE_EABI_TOOLCHAIN_PATH)
    message(FATAL_ERROR "The path to the toolchain (arm-none-eabi) must be specified: -DARM_NONE_EABI_TOOLCHAIN_PATH=<path>")
endif()

if (NOT NRF5_SDK_PATH)
    message(FATAL_ERROR "The path to the NRF52 SDK must be specified: -DNRF5_SDK_PATH=<path>")
endif()

# Set default programmer
if(NOT USE_JLINK AND NOT USE_GDB_CLIENT AND NOT USE_OPENOCD)
    set(USE_JLINK true)
endif()

# Programmer validation
if(USE_JLINK)
    if (NOT NRFJPROG)
        message(FATAL_ERROR "The path to nrfjprog must be specified: -DNRFJPROG=<path>")
    endif()
endif()

if(USE_GDB_CLIENT)
    if(NOT GDB_CLIENT_BIN_PATH)
        set(GDB_CLIENT_BIN_PATH "arm-none-eabi-gdb")
    endif()
    if(NOT GDB_CLIENT_TARGET_REMOTE)
        message(FATAL_ERROR "The GDB target must be specified: -DGDB_CLIENT_TARGET_REMOTE=<target>")
    endif()
endif()

if(USE_OPENOCD)
    if(NOT OPENOCD_BIN_PATH)
        set(OPENOCD_BIN_PATH "openocd")
    endif()
endif()

# Optional features
if(DEFINED USE_DEBUG_PINS AND USE_DEBUG_PINS)
    add_definitions(-DUSE_DEBUG_PINS)
endif()

if(BUILD_DFU)
    set(BUILD_DFU true)
endif()

if(BUILD_RESOURCES)
    set(BUILD_RESOURCES true)
endif()

# Git version tracking
set(PROJECT_GIT_COMMIT_HASH "")
execute_process(
    COMMAND git rev-parse --short HEAD
    WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
    OUTPUT_VARIABLE PROJECT_GIT_COMMIT_HASH
    RESULT_VARIABLE PROJECT_GIT_COMMIT_HASH_SUCCESS
)
string(STRIP "${PROJECT_GIT_COMMIT_HASH}" PROJECT_GIT_COMMIT_HASH)

# Build configuration summary
message("=====================================")
message("  PineTime Build Configuration")
message("=====================================")
message("  Version        : ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}.${PROJECT_VERSION_PATCH}")
message("  Git Commit     : ${PROJECT_GIT_COMMIT_HASH}")
message("  Toolchain      : ${ARM_NONE_EABI_TOOLCHAIN_PATH}")
message("  NRF52 SDK      : ${NRF5_SDK_PATH}")
message("  Build DFU      : ${BUILD_DFU}")
message("  Build Resources: ${BUILD_RESOURCES}")
message("  Debug Pins     : ${USE_DEBUG_PINS}")

if(USE_JLINK)
    message("  Programmer     : JLINK (${NRFJPROG})")
elseif(USE_GDB_CLIENT)
    message("  Programmer     : GDB Client (${GDB_CLIENT_BIN_PATH})")
elseif(USE_OPENOCD)
    message("  Programmer     : OpenOCD (${OPENOCD_BIN_PATH})")
endif()
message("=====================================")

# Version file generation
set(VERSION_EDIT_WARNING "// Do not edit - auto-generated by CMAKE!")
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/Version.h.in ${CMAKE_CURRENT_BINARY_DIR}/Version.h)

# Include toolchain and build configuration
include("cmake-nRF5x/arm-gcc-toolchain.cmake")
include("cmake-nRF5x/nrf52-sdk.cmake")

# Add source subdirectories
add_subdirectory(bootloader)
add_subdirectory(libs)
add_subdirectory(components)
add_subdirectory(displayapp)
add_subdirectory(systemtask)
add_subdirectory(drivers)
add_subdirectory(heartratetask)

# Main application sources
set(MAIN_SRC
    main.cpp
    logging/NrfLogger.cpp
    FreeRTOS/port.c
    FreeRTOS/port_cmsis.c
    FreeRTOS/port_cmsis_systick.c
)

# Create main executable - ONLY IF SOURCES EXIST
if(MAIN_SRC)
    add_executable(pinetime-app ${MAIN_SRC})
    target_link_libraries(pinetime-app PRIVATE
        nrf-sdk
        freertos
        lvgl
        components
        drivers
        systemtask
    )
    set_target_properties(pinetime-app PROPERTIES
        SUFFIX ".out"
        LINK_FLAGS "-Wl,-Map=${CMAKE_CURRENT_BINARY_DIR}/pinetime-app.map"
    )
else()
    message(WARNING "Main sources not defined - skipping pinetime-app target")
endif()

# MCUBoot application
if(BUILD_DFU AND MAIN_SRC)
    add_executable(pinetime-mcuboot-app ${MAIN_SRC})
    target_link_libraries(pinetime-mcuboot-app PRIVATE
        nrf-sdk
        freertos
        lvgl
        components
        drivers
        systemtask
    )
    target_compile_definitions(pinetime-mcuboot-app PRIVATE
        PINETIME_IS_RECOVERY
    )
    set_target_properties(pinetime-mcuboot-app PROPERTIES
        SUFFIX ".out"
        LINK_FLAGS "-Wl,-Map=${CMAKE_CURRENT_BINARY_DIR}/pinetime-mcuboot-app.map"
    )
endif()

# Recovery application
set(RECOVERY_SRC
    recoveryLoader.cpp
)

if(RECOVERY_SRC)
    add_executable(pinetime-recovery-loader ${RECOVERY_SRC})
    target_link_libraries(pinetime-recovery-loader PRIVATE nrf-sdk)
    set_target_properties(pinetime-recovery-loader PROPERTIES
        SUFFIX ".out"
        LINK_FLAGS "-Wl,-Map=${CMAKE_CURRENT_BINARY_DIR}/pinetime-recovery-loader.map"
    )
endif()

# Post-build commands - ONLY FOR VALID TARGETS
if(TARGET pinetime-app)
    add_custom_command(TARGET pinetime-app POST_BUILD
        COMMAND ${CMAKE_SIZE} $<TARGET_FILE:pinetime-app>
        COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:pinetime-app> ${CMAKE_CURRENT_BINARY_DIR}/pinetime-app.bin
        COMMAND ${CMAKE_OBJCOPY} -O ihex $<TARGET_FILE:pinetime-app> ${CMAKE_CURRENT_BINARY_DIR}/pinetime-app.hex
        COMMENT "Building binary and hex files"
    )
endif()

if(TARGET pinetime-mcuboot-app AND BUILD_DFU)
    add_custom_command(TARGET pinetime-mcuboot-app POST_BUILD
        COMMAND ${CMAKE_SIZE} $<TARGET_FILE:pinetime-mcuboot-app>
        COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:pinetime-mcuboot-app> ${CMAKE_CURRENT_BINARY_DIR}/pinetime-mcuboot-app.bin
        COMMAND ${CMAKE_OBJCOPY} -O ihex $<TARGET_FILE:pinetime-mcuboot-app> ${CMAKE_CURRENT_BINARY_DIR}/pinetime-mcuboot-app.hex
        COMMENT "Building MCUBoot binary and hex files"
    )
endif()

# Build resources if requested
if(BUILD_RESOURCES)
    add_subdirectory(displayapp/fonts)
    add_subdirectory(resources)
endif()

message(STATUS "CMake configuration complete")
